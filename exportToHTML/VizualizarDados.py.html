import pandas as pd
import matplotlib.pyplot as plt
import os

# Caminho completo para o arquivo na Área de Trabalho
caminho_arquivo = os.path.join(os.path.expanduser("~"), "Desktop", "aerogerador.dat")

try:
    data = pd.read_csv(caminho_arquivo, sep='\t', header=None)
    data.columns = ['velocidade', 'potencia']  # Definindo as colunas manualmente

    print(data.head())  # Confirmando o carregamento correto

    # Criando o Gráfico de Dispersão
    plt.figure(figsize=(10, 6))
    plt.scatter(data['velocidade'], data['potencia'], color='b', alpha=0.5)
    plt.title("Relação entre Velocidade do Vento e Potência Gerada")
    plt.xlabel("Velocidade do Vento (m/s)")
    plt.ylabel("Potência Gerada (kW)")
    plt.grid(True)
    plt.show()

except FileNotFoundError: #Arquivo na Area de Trabalho
    print("Arquivo 'aerogerador.dat' não encontrado na Área de Trabalho.")
except KeyError as e:
    print(f"Erro de coluna: {e}. Verifique os nomes das colunas.")
